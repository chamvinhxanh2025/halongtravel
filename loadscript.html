cách 1 dùng defer (khuyên dùng) script sẽ tải sau khi load dom hoàn tất

<script src="path_to_file.js" defer></script>

cách 2 dùng DOMContentLoaded hoặc window.onload nếu cần chờ Css, ảnh 

            
            <script>
                document.addEventListener('DOMContentLoaded', () => {
                // Code chạy sau khi DOM sẵn sàng
                console.log('DOM loaded');
                });
            </script>

cách 3 chèn tự động bằng js

            const script = document.createElement('script');
            script.src = 'app.js';
            script.defer = true; // hoặc chờ DOM rồi append
            document.head.appendChild(script);
